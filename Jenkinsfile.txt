pipeline {
     agent any
     stages {
        stage('Restore'){
            steps {
               bat 'dotnet restore HouseRentingsystem.sln'
            }
        }  
        stage('Build') {
         steps {
             bat 'dotnet build HouseRentingSystem.sln --configure Release'
            }
        }
        stage('Test') {
         steps {
             bat 'dotnet test HouseRentingSystem.Tests/HouseRentingSystem.Tests.csproj'
            }
        }
    }
}
